package auth

import (
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"time"
)

// TokenData holds OAuth2 tokens for a single SAP system.
type TokenData struct {
	AccessToken  string    `json:"access_token"`
	RefreshToken string    `json:"refresh_token"`
	ExpiresAt    time.Time `json:"expires_at"`
}

// IsExpired returns true if the access token has expired.
func (t TokenData) IsExpired() bool {
	return time.Now().After(t.ExpiresAt)
}

// TokenStore manages per-system OAuth2 tokens in a JSON file.
type TokenStore struct {
	path string
}

// NewTokenStore creates a TokenStore that reads/writes the given file path.
func NewTokenStore(path string) *TokenStore {
	return &TokenStore{path: path}
}

// DefaultTokenPath returns the default token file path: ~/.config/mcp-server-abap/tokens.json
func DefaultTokenPath() string {
	home, err := os.UserHomeDir()
	if err != nil {
		home = "."
	}
	return filepath.Join(home, ".config", "mcp-server-abap", "tokens.json")
}

// load reads all tokens from disk. Returns empty map if file does not exist.
func (s *TokenStore) load() (map[string]TokenData, error) {
	data, err := os.ReadFile(s.path)
	if err != nil {
		if os.IsNotExist(err) {
			return make(map[string]TokenData), nil
		}
		return nil, fmt.Errorf("reading token file: %w", err)
	}

	var tokens map[string]TokenData
	if err := json.Unmarshal(data, &tokens); err != nil {
		return nil, fmt.Errorf("parsing token file: %w", err)
	}
	return tokens, nil
}

// Save writes a token for the given system, preserving other systems' tokens.
func (s *TokenStore) Save(system string, token TokenData) error {
	tokens, err := s.load()
	if err != nil {
		// If file is corrupt, start fresh
		tokens = make(map[string]TokenData)
	}
	tokens[system] = token

	data, err := json.MarshalIndent(tokens, "", "  ")
	if err != nil {
		return fmt.Errorf("marshaling tokens: %w", err)
	}

	if err := os.MkdirAll(filepath.Dir(s.path), 0700); err != nil {
		return fmt.Errorf("creating token directory: %w", err)
	}

	return os.WriteFile(s.path, data, 0600)
}

// TokenForSystem returns the stored token for the given system name.
func (s *TokenStore) TokenForSystem(system string) (TokenData, error) {
	tokens, err := s.load()
	if err != nil {
		return TokenData{}, err
	}

	token, ok := tokens[system]
	if !ok {
		return TokenData{}, fmt.Errorf("no token found for system %q. Run: mcp-server-abap login %s", system, system)
	}
	return token, nil
}

package auth

import (
	"crypto/rand"
	"crypto/sha256"
	"crypto/tls"
	"encoding/base64"
	"encoding/json"
	"fmt"
	"io"
	"net/http"
	"net/url"
	"time"
)

const (
	// SAP OAuth2 endpoints (appended to host)
	AuthorizePath = "/sap/bc/sec/oauth2/authorize"
	TokenPath     = "/sap/bc/sec/oauth2/token"
)

// GeneratePKCE creates a code_verifier and its S256 code_challenge per RFC 7636.
func GeneratePKCE() (verifier, challenge string, err error) {
	// 32 bytes -> 43 chars base64url (without padding)
	buf := make([]byte, 32)
	if _, err := rand.Read(buf); err != nil {
		return "", "", fmt.Errorf("generating random bytes: %w", err)
	}
	verifier = base64.RawURLEncoding.EncodeToString(buf)

	h := sha256.Sum256([]byte(verifier))
	challenge = base64.RawURLEncoding.EncodeToString(h[:])
	return verifier, challenge, nil
}

// AuthorizeURL builds the SAP OAuth2 authorization endpoint URL.
func AuthorizeURL(host, clientID, redirectURI, codeChallenge string) string {
	params := url.Values{
		"response_type":         {"code"},
		"client_id":             {clientID},
		"redirect_uri":          {redirectURI},
		"code_challenge":        {codeChallenge},
		"code_challenge_method": {"S256"},
	}
	return host + AuthorizePath + "?" + params.Encode()
}

// tokenResponse is the JSON response from the SAP OAuth2 token endpoint.
type tokenResponse struct {
	AccessToken  string `json:"access_token"`
	RefreshToken string `json:"refresh_token"`
	ExpiresIn    int    `json:"expires_in"`
	Error        string `json:"error"`
}

func newHTTPClient(tlsSkipVerify bool) *http.Client {
	return &http.Client{
		Timeout: 30 * time.Second,
		Transport: &http.Transport{
			TLSClientConfig: &tls.Config{
				InsecureSkipVerify: tlsSkipVerify, //nolint:gosec
			},
		},
	}
}

// ExchangeCode exchanges an authorization code for tokens.
func ExchangeCode(host, clientID, code, codeVerifier, redirectURI string, tlsSkipVerify bool) (TokenData, error) {
	form := url.Values{
		"grant_type":    {"authorization_code"},
		"code":          {code},
		"code_verifier": {codeVerifier},
		"client_id":     {clientID},
		"redirect_uri":  {redirectURI},
	}

	return postTokenRequest(host, form, tlsSkipVerify)
}

// RefreshToken exchanges a refresh token for new tokens.
func RefreshToken(host, clientID, refreshToken string, tlsSkipVerify bool) (TokenData, error) {
	form := url.Values{
		"grant_type":    {"refresh_token"},
		"refresh_token": {refreshToken},
		"client_id":     {clientID},
	}

	return postTokenRequest(host, form, tlsSkipVerify)
}

func postTokenRequest(host string, form url.Values, tlsSkipVerify bool) (TokenData, error) {
	client := newHTTPClient(tlsSkipVerify)

	resp, err := client.PostForm(host+TokenPath, form)
	if err != nil {
		return TokenData{}, fmt.Errorf("token request: %w", err)
	}
	defer func() { _ = resp.Body.Close() }()

	body, err := io.ReadAll(resp.Body)
	if err != nil {
		return TokenData{}, fmt.Errorf("reading token response: %w", err)
	}

	if resp.StatusCode != http.StatusOK {
		return TokenData{}, fmt.Errorf("token endpoint returned %d: %s", resp.StatusCode, string(body))
	}

	var tokenResp tokenResponse
	if err := json.Unmarshal(body, &tokenResp); err != nil {
		return TokenData{}, fmt.Errorf("parsing token response: %w", err)
	}

	if tokenResp.Error != "" {
		return TokenData{}, fmt.Errorf("OAuth2 error: %s", tokenResp.Error)
	}

	return TokenData{
		AccessToken:  tokenResp.AccessToken,
		RefreshToken: tokenResp.RefreshToken,
		ExpiresAt:    time.Now().Add(time.Duration(tokenResp.ExpiresIn) * time.Second),
	}, nil
}

package cmd

import (
	"context"
	"fmt"
	"net"
	"net/http"
	"os"
	"os/exec"
	"runtime"
	"time"

	"github.com/dachner/mcp-server-abap/auth"
	"github.com/dachner/mcp-server-abap/config"
)

// ResolveSystem returns the system name and config for the given name (or default).
func ResolveSystem(cfg *config.Config, name string) (string, config.SAPConfig, error) {
	if name == "" {
		name = cfg.DefaultSystem
	}
	sys, ok := cfg.Systems[name]
	if !ok {
		return "", config.SAPConfig{}, fmt.Errorf("system %q not found in config", name)
	}
	return name, sys, nil
}

// ValidateOAuth2System checks that the system is configured for OAuth2 (no user/password).
func ValidateOAuth2System(sys config.SAPConfig, name string) error {
	if !sys.IsOAuth2() {
		return fmt.Errorf("system %q uses Basic Auth (user/password configured). Login is only for OAuth2 systems", name)
	}
	return nil
}

// StartCallbackServer starts a local HTTP server that listens for the OAuth2 callback.
// Returns the port, a shutdown function, and any error.
func StartCallbackServer(ctx context.Context, codeCh chan<- string, errCh chan<- error) (int, func(), error) {
	listener, err := net.Listen("tcp", "localhost:0")
	if err != nil {
		return 0, nil, fmt.Errorf("starting callback server: %w", err)
	}
	port := listener.Addr().(*net.TCPAddr).Port

	mux := http.NewServeMux()
	srv := &http.Server{Handler: mux}

	mux.HandleFunc("/callback", func(w http.ResponseWriter, r *http.Request) {
		code := r.URL.Query().Get("code")
		if code == "" {
			errCh <- fmt.Errorf("no authorization code in callback")
			http.Error(w, "Missing authorization code", http.StatusBadRequest)
			return
		}
		w.Header().Set("Content-Type", "text/html")
		fmt.Fprint(w, "<html><body><h1>Login successful!</h1><p>You can close this window.</p></body></html>")
		codeCh <- code
	})

	go func() {
		if err := srv.Serve(listener); err != nil && err != http.ErrServerClosed {
			errCh <- err
		}
	}()

	shutdown := func() {
		shutdownCtx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
		defer cancel()
		_ = srv.Shutdown(shutdownCtx)
	}

	return port, shutdown, nil
}

// ExchangeAndSave exchanges an auth code for tokens and saves them.
func ExchangeAndSave(sys config.SAPConfig, systemName, code, verifier, redirectURI, tokenPath string) error {
	token, err := auth.ExchangeCode(sys.Host, sys.OAuth2ClientID, code, verifier, redirectURI, sys.TLSSkipVerify)
	if err != nil {
		return fmt.Errorf("exchanging code for token: %w", err)
	}

	store := auth.NewTokenStore(tokenPath)
	if err := store.Save(systemName, token); err != nil {
		return fmt.Errorf("saving token: %w", err)
	}
	return nil
}

// openBrowser opens the given URL in the default browser.
func openBrowser(url string) error {
	switch runtime.GOOS {
	case "linux":
		return exec.Command("xdg-open", url).Start()
	case "windows":
		return exec.Command("rundll32", "url.dll,FileProtocolHandler", url).Start()
	case "darwin":
		return exec.Command("open", url).Start()
	default:
		return fmt.Errorf("unsupported platform %q", runtime.GOOS)
	}
}

// RunLogin executes the full login flow for the given system.
func RunLogin(cfg *config.Config, systemArg string) error {
	name, sys, err := ResolveSystem(cfg, systemArg)
	if err != nil {
		return err
	}
	if err := ValidateOAuth2System(sys, name); err != nil {
		return err
	}

	verifier, challenge, err := auth.GeneratePKCE()
	if err != nil {
		return err
	}

	ctx, cancel := context.WithTimeout(context.Background(), 120*time.Second)
	defer cancel()

	codeCh := make(chan string, 1)
	errCh := make(chan error, 1)

	port, shutdown, err := StartCallbackServer(ctx, codeCh, errCh)
	if err != nil {
		return err
	}
	defer shutdown()

	redirectURI := fmt.Sprintf("http://localhost:%d/callback", port)
	authorizeURL := auth.AuthorizeURL(sys.Host, sys.OAuth2ClientID, redirectURI, challenge)

	fmt.Fprintf(os.Stderr, "Opening browser for login...\n")
	fmt.Fprintf(os.Stderr, "If the browser does not open, visit:\n%s\n", authorizeURL)

	if err := openBrowser(authorizeURL); err != nil {
		fmt.Fprintf(os.Stderr, "Warning: could not open browser: %v\n", err)
	}

	var code string
	select {
	case code = <-codeCh:
	case err := <-errCh:
		return fmt.Errorf("callback error: %w", err)
	case <-ctx.Done():
		return fmt.Errorf("login timed out after 120 seconds")
	}

	tokenPath := auth.DefaultTokenPath()
	if err := ExchangeAndSave(sys, name, code, verifier, redirectURI, tokenPath); err != nil {
		return err
	}

	fmt.Fprintf(os.Stderr, "Login successful for system %q. Token saved.\n", name)
	return nil
}
